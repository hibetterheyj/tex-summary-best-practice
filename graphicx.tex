1. Splitting subfigure across multiple pages

\begin{lstlisting}[language=Tex]
\begin{figure}[b]
    \begin{subfigure}[t]{0.45\hsize}
        \includegraphics[width=0.95\linewidth]{example-image}
        \caption{}
    \end{subfigure}   
    \begin{subfigure}[t]{0.45\hsize}
        \includegraphics[width=0.95\linewidth]{example-image}
        \caption{}
    \end{subfigure}
    \caption{first part of my figure}
\end{figure}

\end{figure}
